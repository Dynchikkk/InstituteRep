Шаг 18 - Функция inet_ntoa().

Конвертирует адрес в стандартный формат интернет: 

char FAR * inet_ntoa
(
	struct in_addr in  
);

Попробуем изменить "Шаг 17 - Функция inet_addr()": 

#include "stdafx.h"
#include "iostream.h"
#include "winsock.h"

#pragma comment(lib,"wsock32.lib")

void main()
{
	in_addr adr1;
	in_addr adr2;
	adr1.s_addr=inet_addr("124.23.45.67");
	adr2.s_addr=inet_addr("as.34.34.56");
	if (adr1.s_addr!=INADDR_NONE) cout << " adr1 correct" << endl;
	else cout << " adr1 incorect " << endl;
	if (adr2.s_addr!=INADDR_NONE) cout << " adr2 correct" << endl;
	else cout << " adr2 incorect" << endl;
	cout << inet_ntoa(adr1) << endl;
	cout << inet_ntoa(adr2) << endl;
}
