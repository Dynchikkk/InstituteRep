start:
    CLR P3.3            ; Display "O"
    SETB P3.4           
    MOV P1, #11000000b  
    MOV P1, #0FFh       
    SETB P3.3           
    CLR P3.4     
       
    JB P2.6, ON         ; If SW6=1, jump to ON

OFF:
    MOV P1, #10001110b  ; Display "FF"
    CLR P3.3            
    MOV P1, #0FFh       
    SJMP start          

ON:
    MOV P1, #10101011b  ; Display "n"
    MOV P1, #0FFh       
    SJMP start
